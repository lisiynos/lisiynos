Двумерные массивы

В двумерной матрице найти цепочку повторяющихся элементов максимальной длины.

#include <conio.h>

int main() {
  // Исходный двумерный массив
  int A[3][5] = {
    {1,2,2,1,2},
    {5,5,5,4,4},
    {7,6,0,1,0}
  };
  int max_c = 0; // Максимальная длина цепочки 
  int max_v = 0; // Числа в максимальной цепочке
  int line = 0; // Номер строки с максимальной цепочкой
  for(int i=0;i<3;i++){
    int v = 0; // Последнее обработанное число 
    int c = 0; // Количество таких чисел (равных v)
    for(int j=0;j<5;j++){
      if(A[i][j] == v){
        c++;
      } else {
        v = A[i][j];
        c = 1;
      }
      if(c > max_c){
        line = i;
        max_c = c;
        max_v = v;
      }
    }
  }
  // Выводим ответ 
  printf("Строка %d - серия %d из %d\n",(line+1),max_v,max_c);
  getch(); // Ожидаем нажатия клавиши перед закрытием программы
  return 0;
}
