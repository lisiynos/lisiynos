Количество простых путей

Простой путь - это путь на клеточном поле, который проходит через каждую клетку не более одного раза.

Нужно вычислить количество возможных простых путей из клетки с координатами $(0,0)$ длины $K$.

#include <cstdlib>
#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

bool used[55][55]; // Клетка использована

long long Answer = 0;
int K;

void R(int x, int y, int t) {
  if (used[x][y]) return;
  if (t == K) { Answer++; return; }
  used[x][y] = true; // visit - посещаем вершину
  // Идём во все стороны
  R(x + 1, y, t + 1); R(x, y + 1, t + 1);     
  R(x - 1, y, t + 1); R(x, y - 1, t + 1);    
  used[x][y] = false; // leave - выходим из вершины
}

#define task  "paths"
int main() {
  freopen(task".in", "r", stdin);
  freopen(task".out", "w", stdout);
  cin >> K;
  for(int i = 0; i < 50; i++)
    for(int j = 0; j < 50; j++) 
      used[i][j] = false;
  R(20, 20, 0);
  cout << Answer;
}
